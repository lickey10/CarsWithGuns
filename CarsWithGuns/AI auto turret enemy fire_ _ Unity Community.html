    Here's a couple scripts i found some time ago that might help you.


    Code (csharp):
    1.  
    2. //Turret Script
    3. #pragma strict
    4.  
    5. var projectile : GameObject;
    6.  
    7. var myTransform : Transform;
    8. var target : Transform;
    9. var turretControl : Transform;
    10. var muzzlePositions : Transform[];
    11.  
    12. var placed : boolean = false;
    13.  
    14. var reloadTime : float = 1;
    15. var turnSpeed : float = 5;
    16. var firePauseTime : float = 0.25;
    17. var errorAmount : float = 1;
    18. var damage : float;
    19. public var speed : float = 3;
    20. private var nextFireTime : float;
    21. private var nextMoveTime : float;
    22. private var aimError : float;
    23.  
    24. private var desiredRotation : Quaternion;
    25.  
    26. private var targetPos : Vector3;
    27.  
    28. function Start ()
    29. {
    30.     myTransform = transform;
    31. }
    32.  
    33. function Update ()
    34. {
    35.     if (target)
    36.     {
    37.         if (Time.time >= nextMoveTime)
    38.         {
    39.             CalculateAimPosition(target.position);
    40.             turretControl.rotation = Quaternion.Lerp(turretControl.rotation, desiredRotation, Time.deltaTime * turnSpeed);
    41.         }
    42.        
    43.         if (Time.time >= nextFireTime)
    44.         {
    45.             FireProjectile();
    46.         }
    47.     }
    48.    
    49.     if (placed == false)
    50.     {
    51.         var distance = transform.position.z - Camera.main.transform.position.z;
    52.         targetPos = new Vector3 (Input.mousePosition.x, Input.mousePosition.y, distance);
    53.         targetPos = Camera.main.ScreenToWorldPoint(targetPos);
    54.     }
    55. }
    56.  
    57. function OnTriggerEnter (other : Collider)
    58. {
    59.     if (other.gameObject.tag == "Player")
    60.     {
    61.         nextFireTime = Time.time + (reloadTime * 0.5);
    62.         target = other.gameObject.transform;
    63.     }
    64. }
    65.  
    66. function OnTriggerExit (other : Collider)
    67. {
    68.     if (other.gameObject.transform == target)
    69.     {
    70.         target = null;
    71.     }
    72. }
    73.  
    74. function CalculateAimPosition(targetPos : Vector3)
    75. {
    76.     var aimPoint = Vector3 (targetPos.x + aimError, target.position.y + aimError, target.position.z + aimError);
    77.     desiredRotation = Quaternion.LookRotation(aimPoint);
    78. }
    79.  
    80. function CalculateAimError()
    81. {
    82.     aimError = Random.Range(-errorAmount, errorAmount);
    83. }
    84.  
    85. function FireProjectile()
    86. {
    87.     audio.Play();
    88.     nextFireTime = Time.time + reloadTime;
    89.     nextMoveTime = Time.time + firePauseTime;
    90.     CalculateAimError();
    91.    
    92.     for(theMuzzlePos in muzzlePositions)
    93.     {
    94.         var newProjectile = Instantiate (projectile, theMuzzlePos.position, theMuzzlePos.rotation);
    95.         var projectileScript : Projectile_BallTurret;
    96.        
    97.         projectileScript = newProjectile.GetComponent(Projectile_BallTurret);
    98.         projectileScript.damage = damage;
    99.     }
    100. }
    101.  
    Code (csharp):
    1.  
    2. //Projectile Script
    3. #pragma strict
    4.  
    5. var speed : float = 10;
    6. var range : float = 10;
    7. var damage : float;
    8.  
    9. private var distance : float;
    10.  
    11.  
    12. function Update ()
    13. {
    14.     transform.Translate(Vector3.forward * Time.deltaTime * speed);
    15.     distance += Time.deltaTime * speed;
    16.    
    17.     if (distance >= range)
    18.     {
    19.         Destroy(gameObject);
    20.     }
    21. }
    22.  
    23. function OnTriggerEnter (other : Collider)
    24. {
    25.     if (other.tag == "Enemy")
    26.     {
    27.         var enemyScript : EnemyScript;
    28.         enemyScript = other.GetComponent(EnemyScript);
    29.        
    30.         enemyScript.ApplyDamage(damage);
    31.        
    32.         Destroy(gameObject);
    33.     }
    34. }
    35.  
    Here's how you set it up:
    1: Create a new prefab.
    2: Create a new sphere.
    3: Add the Projectile Script to the sphere.
    4: Add a rigidbody to the sphere.
    5: Drag the sphere on to the prefab.

    6: Click the new prefab.
    7: Set the variables: Speed, Range and Damage to whatever you like.

    You now have your projectile prefab ready.


    Now to create the turret:
    1: Create a new cube.
    2: Create a new sphere.
    3: Create a new cylinder.
    4: Drag the sphere onto the cube, and position it so you think it looks good.
    5: Drag the cylinder onto the sphere and position it so it looks like a barrel.
    6: Drag the Turret Script on to the cube.
    7: Click the cube, and drag the sphere onto the "turret control" variable.
    8: Drag the projectile prefab from before onto the "projectile" variable.
    9: Extend the "Muzzle Positions" type 1 in the "size" field.
    10: Drag the cylinder onto the new spot in the "Muzzle Positions".
    11: Change the numbers in reloadTime, TurnSpeed, FirePauseTime, ErrorAmount, Damage and Speed to whatever you like.
    12: Add a sphere collider component to the cube, and tick it as "Is Trigger".
    13: Create a new prefab and drag the cube onto that prefab.

    Now it should all work.

    Now if an object with the tag "Player" enters the range of the turret, it will start firing at it.

    Hope this helps
